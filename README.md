Public Function SANJOY_NATH_STRING_TO_CHAR_ARRAY(GGG As String) As String()
        Dim buff() As String
        ReDim buff(Len(GGG) - 1)
        For i = 1 To Len(GGG)
            buff(i - 1) = Mid$(GGG, i, 1)
        Next
        
        STRING_TO_CHAR_ARRAY = buff
End Function 'Public Function SANJOY_NATH_STRING_TO_CHAR_ARRAY(GGG As String) As Characters()

Public Function SANJOY_NATH_CONVERT_DECIMAL_NUMBERS_TO_ASC_DESC_NUMBERS(DECIMAL_NUMBER_STRING As String) As String

 Dim DIGITS_ARR() As String
 
 Dim TEMP_ASC_DESC_NUMBER As String
 
  DIGITS_ARR = SANJOY_NATH_STRING_TO_CHAR_ARRAY(DECIMAL_NUMBER_STRING)
  
  TEMP_ASC_DESC_NUMBER = ""
  TEMP_ASC_DESC_NUMBER = SANJOY_NATH_CONVERT_FIRST_OR_AFTER_DOT_DECIMAL_DIGIT_TO_ASC_DESC_DIGIT(DIGITS_ARR(0)) 'THIS LOGIC HOLDS FOR FIRST DIGIT OR FOR THE DIGIT JUST AFTER DECIMAL DOT IN THE NUMBER
  
    For R = 1 To (UBound(DIGITS_ARR)) Step 1
     
       TEMP_ASC_DESC_NUMBER = Trim(TEMP_ASC_DESC_NUMBER & Trim(SANJOY_NATH_CONVERT_INTERMEDIATE_DECIMAL_DIGIT_TO_JUMP_ASC_DESC_DIGIT(Val(DIGITS_ARR(R)) - Val(DIGITS_ARR(R - 1)))))
    
    Next R 'For R = 0 To (UBound(DIGITS_ARR) - 1) Step 1
  
  
  SANJOY_NATH_CONVERT_DECIMAL_NUMBERS_TO_ASC_DESC_NUMBERS = TEMP_ASC_DESC_NUMBER
  
  
End Function 'Public Function SANJOY_NATH_CONVERT_DECIMAL_NUMBERS_TO_ASC_DESC_NUMBERS(NUMBER As String) As String

Public Function SANJOY_NATH_CONVERT_ASC_DESC_NUMBERS_TO_DECIMAL_NUMBERS_TAKING_CARE_OF_CUMULATION_STATES(ASC_DESC_NUMBER As String) As String


'WE HAVE CHECKED THAT THE SYSTEM WITHOUT CUMULATIVE STATE CHECK THE SYSTEM WORKED FINE UPTO 2 CHARACTER ASCEND DESCEND NUMBERS
'BUT THE CONVERSION SYSTEM FAILED FOR 3 OR MORE CHARACTER ASCEND DESCEND NUMBERS WHICH HAS GIVEN US INSIGHTS REGARDING MORE PARADOXES THAT CAN OCCUR
'SO WE NEED TO REFINE OUR AXIOMS FOR THIS NEW NUMBER SYSTEM

 Dim ALPHA_ARR() As String
 
 Dim TEMP_ASC_DESC_NUMBER As String
 
 Dim VALUE_UPTO_CURRENT_STATE_FROM_LEFT_SIDE_CUMULATION As Double
 
 
  ALPHA_ARR = SANJOY_NATH_STRING_TO_CHAR_ARRAY(ASC_DESC_NUMBER)
  
  'FOR A NUMBER WHOSE LENGTH IS N CAN HAVE (10^N) -1 NUMBER OF NUMBERS.WE KNOW VALUES OF ALL THESE NUMBERS.THESE ARE ALL VALID NUMBERS.
  'BUT FOR ASCEND-DESCEND NUMBERS , THERE ARE 19 POSSIBLE DIGITS AND SO LENGTH OF N MEANS (19^N)-1 POSSIBLE NUMBERS (PERMUTATIONS) ARE THERE IN ASCEND DESCEND NUMBERS SYSTEM.
  'OUT OF ALL THESE  POSSIBLE ARRANGEMENTS OF ASCEND DESCEND NUMBERS , ONLY (10^N)-1 NUMBERS CAN HAVE VALUES(N ALPHA DIGIT ASCEND DESCEND NUMBER CORRESPONDS TO N DIGIT DECIMAL NUMBER(AXIOM OF ASCEND DESCEND NUMBER SYSTEMS)
  'THIS SIGNIFIES THAT ((19^N)-1) -((10^N)-1) NUMBERS OF INVALID NUMBERS(WHOSE VALUATION IS NOT POSSIBLE IN DECIMAL SYSTEM NOR HAS ANY COUNTABLE/MEASURABLE VALUE OTHER THAN SOME INFORMATION VALUE , CRYPTIC VALUE USEFUL FOR SOME OTHER PURPOSE OTHER THAN REGULAR ARITHMATIC OPERATIONS)
  '(19^N - 10^N ) INVALID NUMBERS ARE POSSIBLE IN ASCEND DESCEND NUMBER SYSTEMS SINCE ALL PERMUTATION OF ASCEND DESCEND ALPHA DIGITS DONT REALLY HAVE VALID FACE VALUE(AXIOM) WHICH MEANS FACE VALUE OF EACH ALPHA DIGIT IN ASCEND DESCEND NUMBER SYSTEM DEPENDS UPON ITS POSITION IN THE NUMBER STRING.
  'CHANCES OF THE VALIDITY OF ANY PERMUTED STRING IN ASCEND DESCEND NUMBER SYSTEM IS GIVEN THROUGH (10^N/19^N ) WHICH DECREASES DRASTICALLY AFTER THE LENGTH OF NUMBERS GREATER THAN 7 THAT MEANS PERMUTATIONS OF ASCEND DESCEND NUMBERS GIVE MORE AND MORE CRYPTIC SCENARIO FOR LARGER NUMBERS
  
  TEMP_DECIMAL_NUMBER = ""
  TEMP_DECIMAL_NUMBER = SANJOY_NATH_CONVERT_FIRST_OR_AFTER_DOT_ALPHABET_DIGIT_TO_DECIMAL_DIGIT(ALPHA_ARR(0)) 'THIS LOGIC HOLDS FOR FIRST ALPHABET_DIGIT OR FOR THE ALPHABET_DIGIT JUST AFTER DECIMAL DOT IN THE ASCENDING DESCENDING NUMBER SYSTEM
  
  VALUE_UPTO_CURRENT_CONTIGUOUS_LEFT_DIGIT_VALUATION_STATE_FROM_LEFT_SIDE_CUMULATION = Val(TEMP_DECIMAL_NUMBER) ' WE HAVE A PLAN TO DESIGN ANOTHER NUMBER SYSTEM WHERE WE WILL CUMULATE FROM RIGHT SIDE.THAT WILL GIVE DIFFERENT STRUCTURE OF CONVERSION , AND ARITHMATIC
  
    For R = 1 To (UBound(ALPHA_ARR)) Step 1
     
     
        VALUE_UPTO_CURRENT_CONTIGUOUS_LEFT_DIGIT_VALUATION_STATE_FROM_LEFT_SIDE_CUMULATION = VALUE_UPTO_CURRENT_CONTIGUOUS_LEFT_DIGIT_VALUATION_STATE_FROM_LEFT_SIDE_CUMULATION + Val(SANJOY_NATH_CONVERT_FIRST_OR_AFTER_DOT_ALPHABET_DIGIT_TO_DECIMAL_DIGIT(ALPHA_ARR(R)))
     
      '(ERROR PRONE STEP FOR  MORE THAN 2 DIGITS CASES SO DISCARDED IN THIS) TEMP_DECIMAL_NUMBER = Trim(TEMP_DECIMAL_NUMBER & Trim(CONVERT_INTERMEDIATE_ASCEND_DESCEND_ALPHABET_DIGIT_TO_DECIMAL_DIGIT(ALPHA_ARR(R), ALPHA_ARR(R - 1))))
       TEMP_DECIMAL_NUMBER = Trim(TEMP_DECIMAL_NUMBER & Trim(VALUE_UPTO_CURRENT_CONTIGUOUS_LEFT_DIGIT_VALUATION_STATE_FROM_LEFT_SIDE_CUMULATION)) ' + Val(CONVERT_FIRST_OR_AFTER_DOT_ALPHABET_DIGIT_TO_DECIMAL_DIGIT(ALPHA_ARR(R))))
      
    Next R 'For R = 0 To (UBound(DIGITS_ARR) - 1) Step 1
  
   ' AS PER DIFINITIONAL AXIOMS OF ASCEND DESCEND NUMBER STRING SYSTEM LENGTH OF DECIMAL NUMBER = LENGTH OF ASCEND DESCEND STRING
  If Len(TEMP_DECIMAL_NUMBER) <> Len(ASC_DESC_NUMBER) Then
   TEMP_DECIMAL_NUMBER = "INVALID_LENGTH"
  Else
  
   SANJOY_NATH_CONVERT_ASC_DESC_NUMBERS_TO_DECIMAL_NUMBERS_TAKING_CARE_OF_CUMULATION_STATES = TEMP_DECIMAL_NUMBER
   
  End If 'If Len(TEMP_DECIMAL_NUMBER) <> Len(ASC_DESC_NUMBER) Then
  
  
  
End Function 'Public Function SANJOY_NATH_CONVERT_ASC_DESC_NUMBERS_TO_DECIMAL_NUMBERS_TAKING_CARE_OF_CUMULATION_STATES(NUMBER As String) As String

'IF THERE IS ANY CONVERSION ERROR THAT IS AFTER JUMP THE VALUE BECOMES GREATER THAN 9 WE WILL CONSIDER THAT AS THE MISSING VOWEL U AND (CURRENT CUMULATIVE VALUE SET TO ZERO 0) FOR THE ERROR AFTER JUMP CAUSES LESS THAN ZERO WE WILL CONSIDER THAT AS u(AND CURRENT CUMULATIVE VALUE SET TO ZERO 0)
'THIS GIVES US ERROR DETECTION WHERE WE HAVE MISTAKENLY ASSIGNED WRONG JUMP VALUES ALPHABET WHILE DESIGNING THE ASCEND-DESCEND NUMBERS ENCODING
'THERE ARE SEVERAL POSSIBLE ENCODING MISTAKES CAN HAPPENBUT WE WILL SET DIFFERENT ALPHABETS OR ASCII SYMBOLS FOR THESE DIFFERENT KIND OF ENCODING ERRORS AND CUMULATIVE RESET VALUE


Dim TEMP_CURRENT_DECIMAL_DIGIT As String


If ASCEND_DESCEND_JUMP_VALUE = "O" Or ASCEND_DESCEND_JUMP_VALUE = "o" Then
